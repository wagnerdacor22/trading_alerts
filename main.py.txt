from flask import Flask, request, jsonify
import requests
import os
from datetime import datetime
import time
import threading

app = Flask(__name__)

# Configurações
TELEGRAM_TOKEN = os.environ.get('TELEGRAM_TOKEN', '')
TELEGRAM_CHAT_ID = os.environ.get('TELEGRAM_CHAT_ID', '')

# Variáveis para controle
ultimo_sinal = None

def enviar_telegram(mensagem):
    url = f"https://api.telegram.org/bot{TELEGRAM_TOKEN}/sendMessage"
    payload = {
        "chat_id": TELEGRAM_CHAT_ID,
        "text": mensagem,
        "parse_mode": "HTML"
    }
    try:
        response = requests.post(url, json=payload, timeout=10)
        return response.json()
    except Exception as e:
        print(f"Erro: {e}")
        return None

def verificar_sinais():
    global ultimo_sinal
    
    # API do TradingView (pública)
    # SOLUSDT na Binance, timeframe 1H
    url = "https://api.binance.com/api/v3/klines"
    params = {
        "symbol": "SOLUSDT",
        "interval": "1h",
        "limit": 50
    }
    
    try:
        response = requests.get(url, params=params, timeout=10)
        data = response.json()
        
        # Calcular EMA 5 e BB (21, 2)
        closes = [float(candle[4]) for candle in data]
        
        # EMA 5
        ema5 = sum(closes[-5:]) / 5
        
        # SMA 21 (BB Central)
        sma21 = sum(closes[-21:]) / 21
        
        # Desvio padrão
        import statistics
        desvio = statistics.stdev(closes[-21:])
        
        # Bandas
        bb_superior = sma21 + (2 * desvio)
        bb_inferior = sma21 - (2 * desvio)
        
        # Verificar cruzamento
        preco_atual = closes[-1]
        preco_anterior = closes[-2]
        ema5_anterior = sum(closes[-6:-1]) / 5
        
        # Compra: EMA 5 cruza ACIMA da BB Central
        if ema5_anterior < sma21 and ema5 >= sma21:
            if ultimo_sinal != "COMPRA":
                mensagem = f"""🟢 <b>SINAL DE COMPRA SOLUSDT</b>

💰 Preço: {preco_atual:.2f}
📈 EMA 5: {ema5:.2f}
📊 BB Central: {sma21:.2f}
⏰ Timeframe: 1H

✅ EMA 5 cruzou ACIMA da BB Central"""
                enviar_telegram(mensagem)
                ultimo_sinal = "COMPRA"
                print(f"Compra enviada: {preco_atual}")
        
        # Venda: EMA 5 cruza ABAIXO da BB Central
        elif ema5_anterior > sma21 and ema5 <= sma21:
            if ultimo_sinal != "VENDA":
                mensagem = f"""🔴 <b>SINAL DE VENDA SOLUSDT</b>

💰 Preço: {preco_atual:.2f}
📈 EMA 5: {ema5:.2f}
📊 BB Central: {sma21:.2f}
⏰ Timeframe: 1H

✅ EMA 5 cruzou ABAIXO da BB Central"""
                enviar_telegram(mensagem)
                ultimo_sinal = "VENDA"
                print(f"Venda enviada: {preco_atual}")
        
    except Exception as e:
        print(f"Erro ao verificar: {e}")

def loop_verificacao():
    while True:
        verificar_sinais()
        # Verifica a cada 5 minutos
        time.sleep(300)

@app.route('/')
def home():
    return "✅ Servidor de Sinais SOLUSDT rodando!"

@app.route('/teste')
def teste():
    mensagem = "🧪 <b>Teste de conexão!</b>\n\nServidor está funcionando! ✅"
    enviar_telegram(mensagem)
    return "Teste enviado!"

@app.route('/verificar')
def verificar():
    verificar_sinais()
    return "Verificação manual executada!"

if __name__ == '__main__':
    # Inicia thread de verificação automática
    thread = threading.Thread(target=loop_verificacao)
    thread.daemon = True
    thread.start()
    
    print("🚀 Servidor iniciado!")
    print("📱 Verificação automática a cada 5 minutos")
    app.run(host='0.0.0.0', port=int(os.environ.get('PORT', 5000)))
